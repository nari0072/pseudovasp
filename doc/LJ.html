def atom_force()
  f=[0.0,0.0,0.0]
  nl.each do |aj|
    x,y,z=f_distance(@pos,aj.pos)
    r=sqrt(x**2+y**2+z**2)
    dedr=((3.0*A/r**4)-(5.0*B/r**6))
    f[0] += -x/r*dedr
    f[1] += -y/r*dedr
    f[2] += -z/r*dedr
  end
  return f
end
