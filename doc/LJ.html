def atom_energy()
  ene=0.0
  @nl.each do |aj|
    r=distance(@pos,aj.pos)
    ene += @@potential.energy(r)
  end
  return ene
end

def atom_force()
  f=[0.0,0.0,0.0]
  @nl.each do |aj|
    x,y,z=f_distance(@pos,aj.pos)
    r=sqrt(x**2+y**2+z**2)
    dedr = @@potential.dedr(r)
    f[0] += -x/r*dedr
    f[1] += -y/r*dedr
    f[2] += -z/r*dedr
  end
  return f
end

def select(file='POTCAR')
  src = YAML.load_file(file)
  @@potential=case src[:type]
            when 'lj0'
              LJ0.new(src)
            when 'lj_jindo'
              LJ_Jindo.new(src)
            end
end
