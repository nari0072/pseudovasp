module LJに記述

  def select(file='POTCAR')
#    p src = YAML.load_file(ARGV[0])
    p src = YAML.load_file(file)
    @@potential=case src[:type]
              when 'lj0'
                LJ0.new(src)
              when 'lj_jindo'
                LJ_Jindo.new(src)
              end
  end
